## Waht about `this`?

`this` is the lexical scope of the function that is being called

```js

  pi: 3.1415,
  area: function(r) {
    return this.pi * r * r;
  }
}

const a = circleUtil.area(10);
console.log(a)
```

```js
    console.log(this.song)
```

## Arrow functions

```js
// Traditional Function
function (a){
  return a + 100;
}

// Arrow Function Break Down

// 1. Remove the word "function" and place arrow between the argument and opening body bracket
(a) => {
  return a + 100;
}

// 2. Remove the body brackets and word "return" -- the return is implied.
(a) => a + 100;

// 3. Remove the argument parentheses
a => a + 100;
```

```js
function unsafeParse(str) {

}

function safeParse(str) {
  try{
    return JSON.parse(str)
  }catch(err) {
    return {};
  }
}

```

## Error Hanedling with callback

```js
function unsafeParse(str, callback) {
  const obj = JSON.parse(str);
  callback(obj);
}

function safeParse(str, callback) {
  try{
    const obj = JSON.parse(str)
    callback(null, obj)
  }catch(err) {
    callback(err);
  }
}

function onParse(error, obj) {
  if (error) {
    console.log(error);
    return;
  }

  // obj is valid
  // do stuff
  // ...
}
```
